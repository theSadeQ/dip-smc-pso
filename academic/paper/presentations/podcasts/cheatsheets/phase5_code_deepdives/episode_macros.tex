% ------------------------------------------------------------------------------
% SIDE-BY-SIDE COMPARISON (Warning vs Tip)
% Example:
% \sidebysidecomparison{Without Interface}{\begin{itemize}...\end{itemize}}
%                     {With Interface}{\begin{itemize}...\end{itemize}}
% ------------------------------------------------------------------------------
\newcommand{\sidebysidecomparison}[4]{%
    \begin{multicols}{2}
    \begin{warning}
    \textbf{#1}\\
    #2
    \end{warning}

    \columnbreak

    \begin{tip}
    \textbf{#3}\\
    #4
    \end{tip}
    \end{multicols}
}

% ------------------------------------------------------------------------------
% GENERIC SIDE-BY-SIDE BOXES (Custom box types)
% Example:
% \sidebysideboxes{warning}{Left Title}{Left Body}{example}{Right Title}{Right Body}
% ------------------------------------------------------------------------------
\newcommand{\sidebysideboxes}[6]{%
    \begin{multicols}{2}
    \begin{#1}
    \textbf{#2}\\
    #3
    \end{#1}

    \columnbreak

    \begin{#4}
    \textbf{#5}\\
    #6
    \end{#4}
    \end{multicols}
}

% ------------------------------------------------------------------------------
% FOUR-KEY-PATTERNS SUMMARY BOX
% Example:
% \fourkeypatterns{Pattern 1}{Pattern 2}{Pattern 3}{Pattern 4}
% ------------------------------------------------------------------------------
\newcommand{\fourkeypatterns}[4]{%
    \begin{tcolorbox}[colback=background, colframe=primary, title=\faLayerGroup\ Four Key Patterns]
    \begin{enumerate}
        \item #1
        \item #2
        \item #3
        \item #4
    \end{enumerate}
    \end{tcolorbox}
}

% ------------------------------------------------------------------------------
% EPISODE CROSS-REFERENCE BOX
% Example:
% \episodexref{E031}{Classical SMC Implementation}{Deep dive into sliding surface design.}
% ------------------------------------------------------------------------------
\newcommand{\episodexref}[3]{%
    \begin{tcolorbox}[colback=background, colframe=secondary, title=\faLink\ Related Episode]
    \textbf{#1}: #2\\
    \textit{#3}
    \end{tcolorbox}
}
