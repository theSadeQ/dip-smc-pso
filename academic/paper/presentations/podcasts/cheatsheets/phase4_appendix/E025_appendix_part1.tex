\section{Configuration System}
\textbf{Central File}: \texttt{config.yaml} (Pydantic-validated)
\subsection{Key Sections}
\begin{itemize}
\item \textbf{physics}: Mass, length, damping, gravity
\item \textbf{controller\_settings}: Gains for 7 SMC variants
\item \textbf{pso\_parameters}: Swarm size (50), iterations (100)
\item \textbf{simulation\_settings}: Time step (0.01s), duration (10s)
\end{itemize}
\section{Installation}
\begin{lstlisting}[style=bashstyle]
git clone https://github.com/theSadeQ/dip-smc-pso
cd dip-smc-pso
python -m venv .venv
source .venv/bin/activate  # Windows: .venv\Scripts\activate
pip install -r requirements.txt
\end{lstlisting}
\section{Quick Start}
\begin{lstlisting}[style=bashstyle]
# Run simulation
python simulate.py --ctrl classical_smc --plot

# Run PSO optimization
python simulate.py --ctrl classical_smc --run-pso --save gains.json

# Launch web UI
streamlit run streamlit_app.py
\end{lstlisting}
