\section{Lessons Learned}

\subsection{Technical Lessons}
\begin{enumerate}
  \item \textbf{Pydantic YAML Validation}: Caught 18 config errors pre-runtime
  \item \textbf{Numba Vectorization}: 20x speedup for batch simulations
  \item \textbf{Weakref Patterns}: Prevented memory leaks in 100K+ step simulations
  \item \textbf{Multi-Agent Orchestration}: Survived 3 token limit events via checkpoints
  \item \textbf{MCP Auto-Trigger}: Reduced manual tool selection by 70\%
\end{enumerate}

\subsection{Process Lessons}
\begin{enumerate}
  \item \textbf{Maintenance Mode}: Frozen UI work enabled LT-7 paper completion
  \item \textbf{Checkpoint System}: Zero work lost across 11 research tasks (46 hours)
  \item \textbf{Quality Gates}: 7/8 gates enforced in pre-commit hooks
  \item \textbf{Documentation Standards}: <5 AI patterns per file via automated detection
  \item \textbf{Testing Philosophy}: 87\% coverage, 96\% critical paths
\end{enumerate}

\subsection{Architectural Lessons}
\begin{enumerate}
  \item \textbf{Intentional Patterns}: Compatibility layers (optimizer/ -> optimization/) prevent breaking changes
  \item \textbf{Re-export Chains}: simulation\_context.py in 3 locations for import flexibility
  \item \textbf{Model Variants}: 8 dynamics files for different accuracy/performance tradeoffs
  \item \textbf{Framework Files}: test\_automation.py is PRODUCTION code, not tests
\end{enumerate}
