166def __getattr__(name): 

167 """ 

168 Lazy import attributes on demand. 

169 

170 This function is called when an attribute is not found in the module's namespace. 

171 It allows us to defer imports until they're actually needed, avoiding loading 

172 heavy dependencies (websockets, pyserial, etc.) at module import time. 

173 """ 

174 if name in _LAZY_IMPORTS: 

175 # Check cache first 

176 if name in _imported_cache: 

177 return _imported_cache[name] 

178 

179 # Import the module and get the attribute 

180 module_name, attr_name = _LAZY_IMPORTS[name] 

181 try: 

182 from importlib import import_module 

183 module = import_module(f'.{module_name}', package='interfaces') 

184 attr = getattr(module, attr_name) 

185 _imported_cache[name] = attr 

186 return attr 

187 except ImportError as e: 

188 raise AttributeError( 

189 f"Cannot import '{name}' from interfaces.{module_name}. " 

190 f"This may be due to missing optional dependencies. " 

191 f"Original error: {e}"
