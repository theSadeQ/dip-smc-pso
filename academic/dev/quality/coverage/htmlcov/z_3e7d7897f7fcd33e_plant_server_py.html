39def _build_dynamics(cfg: dict): 

40 """ 

41 Build the dynamics model using the project's light/full models if present. 

42 """ 

43 use_full = bool(_get(cfg, "simulation.use_full_dynamics", False)) 

44 try: 

45 if use_full: 

46 from src.plant.models.full.full_dynamics import FullDynamics as Model 

47 else: 

48 from src.plant.models.simplified.simplified_dynamics import SimplifiedDynamics as Model 

49 except Exception: 

50 class Model: 

51 def __init__(self, dt: float): 

52 self.dt = dt 

53 def step(self, x, u): 

54 x = x.copy() 

55 x[0] += self.dt 

56 x[1] += self.dt * x[3] 

57 x[2] += self.dt * x[4] 

58 x[3] += self.dt * (u if np.isscalar(u) else u[0] - 0.1 * x[3]) 

59 x[4] += self.dt * ((0.0 if np.isscalar(u) else u[1]) - 0.1 * x[4]) 

60 x[5] += 0.0 

61 return x 

62 return Model 

63 return Model

65class PlantServer: 

66 # Network packet formats incorporating sequence numbers and CRC‑32. 

67 # Each command from the controller includes an unsigned 32‑bit sequence 

68 # number, a 64‑bit float command value and a CRC computed over the 

69 # sequence and command. Similarly, each state packet sent back to 

70 # the controller contains a sequence number, six 64‑bit floats and a 

71 # CRC. Including a sequence number enables detection of out‑of‑order 

72 # or duplicate packets, while the CRC provides integrity checking 

73 # against bit errors【401883805680716†L137-L149】. See design review issue #44. 

74 CMD_FMT = "!I d I" # [CIT-066] 

75 CMD_SIZE = struct.calcsize(CMD_FMT) 

76 STATE_FMT = "!I 6d I" # [CIT-066] 

77 STATE_SIZE = struct.calcsize(STATE_FMT)

92 self.bind_addr = bind_addr 

93 self.dt = float(dt) 

94 self.extra_latency = float(extra_latency_ms) / 1000.0 

95 # Standard deviation of additive sensor noise. A zero value 

96 # disables noise injection. When non‑zero the noise is 

97 # generated using the instance's random number generator 

98 # ``self.rng`` rather than the global NumPy RNG. This design 

99 # follows best practices in stochastic simulation by avoiding 

100 # global state and enabling reproducible noise sequences when a 

101 # seed is provided【474685546011607†L282-L292】. 

102 self.sensor_noise_std = float(sensor_noise_std) 

103 self.max_steps = max_steps 

104 self._ready_evt = server_ready_event 

105 

106 self._sock: Optional[socket.socket] = None 

107 self._stop_evt = threading.Event() 

108 

109 init = _get(cfg, "simulation.initial_state", [0, 0, 0, 0, 0, 0]) 

110 self.state = np.array(init, dtype=float) 

111 

112 Model = _build_dynamics(cfg) 

113 self.model = Model(self.dt) 

114 

115 # Random number generator for sensor noise. If a generator is 

116 # provided by the caller use it directly; otherwise, create 

117 # a new generator seeded from the configuration if available. 

118 if rng is not None: 

119 self.rng = rng 

120 else: 

121 # Attempt to read a seed from the configuration. Use 

122 # ``global_seed`` if present or fall back to None for 

123 # non‑deterministic noise. Storing the seed at the 

124 # application level centralises randomness control across 

125 # subsystems and avoids hidden correlations【474685546011607†L282-L292】. 

126 seed = _get(cfg, "global_seed", None) 

127 if seed is not None: 

128 try: 

129 self.rng = np.random.default_rng(int(seed)) 

130 except Exception: 

131 self.rng = np.random.default_rng() 

132 else: 

133 self.rng = np.random.default_rng()

135 # Sequence numbers for command reception and state transmission. The 

136 # plant server increments ``rx_seq`` when processing commands and 

137 # echoes back the last received sequence in the measurement. This 

138 # facilitates detection of stale or dropped packets and allows 

139 # simple acknowledgement semantics. See design review issue #44. 

140 self.rx_seq: int = 0 

141 self.tx_seq: int = 0

143 def start(self) -> None: 

144 try: 

145 self._sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM) 

146 self._sock.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1) 

147 self._sock.bind(self.bind_addr) 

148 self._sock.settimeout(1.0) 

149 # Signal readiness AFTER successful bind 

150 if self._ready_evt is not None: 

151 self._ready_evt.set() 

152 except Exception: 

153 # Unblock waiter and exit if bind fails 

154 if self._ready_evt is not None: 

155 self._ready_evt.set() 

156 logging.exception("PlantServer failed to bind/start") 

157 return 

158 

159 steps = 0 

160 try: 

161 while not self._stop_evt.is_set(): 

162 try: 

163 data, client_addr = self._sock.recvfrom(self.CMD_SIZE) 

164 except socket.timeout: 

165 continue 

166 if not data or len(data) < self.CMD_SIZE: 

167 continue 

168 # Decode the command packet: sequence, command, CRC. 

169 try: 

170 seq, cmd_val, recv_crc = struct.unpack(self.CMD_FMT, data[: self.CMD_SIZE]) 

171 except Exception: 

172 continue 

173 # Compute CRC over the sequence and command value (excluding the CRC field). 

174 crc_payload = struct.pack("!I d", seq, cmd_val) # [CIT-066] 

175 calc_crc = zlib.crc32(crc_payload) & 0xFFFFFFFF # [CIT-066] 

176 # Ignore packets with mismatched CRC or stale sequence numbers. 

177 if recv_crc != calc_crc or seq < self.rx_seq: 

178 continue 

179 # Update the last received sequence. 

180 self.rx_seq = seq 

181 u = float(cmd_val) 

182 # Step the dynamics. Accept both scalar and vector commands. 

183 try: 

184 self.state = self.model.step(self.state, u) 

185 except Exception: 

186 try: 

187 self.state = self.model.step(self.state, np.array([u, 0.0], dtype=float)) 

188 except Exception: 

189 pass 

190 # Copy state and inject noise. 

191 meas = self.state.copy() 

192 if self.sensor_noise_std > 0.0: 

193 meas += self.rng.normal(0.0, self.sensor_noise_std, size=meas.shape) 

194 # Optional latency. 

195 if self.extra_latency > 0.0: 

196 time.sleep(self.extra_latency) 

197 # Compose the state packet with sequence number and CRC. Echo the 

198 # last received sequence back to the controller as acknowledgement. 

199 self.tx_seq = self.rx_seq 

200 # Pack sequence and measurement values (6 floats). 

201 payload = struct.pack("!I 6d", self.tx_seq, *[float(v) for v in meas[:6]]) # [CIT-066] 

202 state_crc = zlib.crc32(payload) & 0xFFFFFFFF # [CIT-066] 

203 pkt = payload + struct.pack("!I", state_crc) # [CIT-066] 

204 try: 

205 self._sock.sendto(pkt, client_addr) 

206 except Exception: 

207 pass 

208 steps += 1 

209 if self.max_steps is not None and steps >= self.max_steps: 

210 break 

211 finally: 

212 try: 

213 self.close() 

214 except Exception: 

215 pass
