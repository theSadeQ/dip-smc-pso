77def create_performance_analyzer(analyzer_type: str = "advanced", **kwargs): 

78 """Create performance analyzer instance. 

79 

80 Parameters 

81 ---------- 

82 analyzer_type : str 

83 Type of analyzer ('advanced', 'stability', 'robustness') 

84 **kwargs 

85 Additional configuration parameters 

86 

87 Returns 

88 ------- 

89 PerformanceAnalyzer 

90 Configured performance analyzer 

91 """ 

92 if analyzer_type == "advanced": 

93 return AdvancedControlMetrics(**kwargs) 

94 elif analyzer_type == "stability": 

95 return StabilityAnalyzer(**kwargs) 

96 elif analyzer_type == "robustness": 

97 return RobustnessAnalyzer(**kwargs) 

98 else: 

99 raise ValueError(f"Unknown analyzer type: {analyzer_type}") 

100
