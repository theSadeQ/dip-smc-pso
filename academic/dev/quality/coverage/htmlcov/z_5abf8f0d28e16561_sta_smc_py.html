34@numba.njit(cache=True) 

35def _sta_smc_control_numba( 

36 state: np.ndarray, 

37 z: float, 

38 alg_gain_K1: float, alg_gain_K2: float, surf_gain_k1: float, surf_gain_k2: float, surf_lam1: float, surf_lam2: float, 

39 damping_gain: float, dt: float, max_force: float, boundary_layer: float, 

40 u_eq: float = 0.0 

41) -> Tuple[float, float, float]: 

42 """ 

43 Numba‑accelerated core of the Super‑Twisting SMC. 

44 

45 Uses a saturated sign function for sigma to maintain full control authority 

46 outside the boundary layer and linear behavior inside it, which is required 

47 for robust, finite‑time convergence of the super‑twisting algorithm. Levant 

48 (2003) introduced the super-twisting controller to achieve second-order 

49 sliding mode with continuous control【smc_levant_2003_higher_order 

50 _introduction†L1-L24】. Moreno & Osorio (2008) provided the first Lyapunov 

51 -based proof of finite-time convergence【smc_moreno_2008_lyapunov_sta†L2856 

52 -L2861】. Seeber & Horn (2017) refined parameter conditions for lower gains 

53 and reduced chattering【smc_seeber_2017_sta_parameter_setting†L241-L243】. 

54 """ 

55 _, th1, th2, _, th1dot, th2dot = state 

56 

57 # Sliding surface: sigma = k1*(th1dot + lam1*th1) + k2*(th2dot + lam2*th2) 

58 sigma = surf_gain_k1 * (th1dot + surf_lam1 * th1) + surf_gain_k2 * (th2dot + surf_lam2 * th2) 

59 

60 # ---- FIX CS‑01: robust saturation instead of "sigma/(|sigma|+eps)" ---- 

61 eps = boundary_layer # guaranteed > 0 by __init__ 

62 if np.abs(sigma) > eps: 

63 sgn_sigma = np.sign(sigma) # full control authority 

64 else: 

65 sgn_sigma = sigma / eps # linear inside boundary layer 

66 # ----------------------------------------------------------------------- 

67 

68 # Super‑twisting continuous term and disturbance‑like internal state 

69 u_cont = -alg_gain_K1 * np.sqrt(np.abs(sigma)) * sgn_sigma 

70 # Use the previous z (integral state) for the disturbance term. In the 

71 # continuous super‑twisting algorithm, the integral term z enters the 

72 # control law directly and is updated after computing the control. The 

73 # earlier implementation incorrectly used the updated z in the same step. 

74 u_dis = z 

75 new_z = z - alg_gain_K2 * sgn_sigma * dt 

76 

77 # Damping on sigma (optional) 

78 u = u_eq + u_cont + u_dis - damping_gain * sigma 

79 

80 # Saturate final outputs to actuator limits. Numba cannot 

81 # compile np.clip on scalars when running in nopython mode. Use 

82 # explicit conditional clamps instead. 

83 if u > max_force: 

84 u = max_force 

85 elif u < -max_force: 

86 u = -max_force 

87 if new_z > max_force: 

88 new_z = max_force 

89 elif new_z < -max_force: 

90 new_z = -max_force 

91 return float(u), float(new_z), float(sigma)

94@numba.njit(cache=True) 

95def _sta_smc_core( 

96 z: float, 

97 sigma: float, 

98 sgn_sigma: float, 

99 alg_gain_K1: float, 

100 alg_gain_K2: float, 

101 damping_gain: float, 

102 dt: float, 

103 max_force: float, 

104 u_eq: float = 0.0, 

105 Kaw: float = 0.0, 

106) -> Tuple[float, float, float]: 

107 """Numba-accelerated core using precomputed sigma and its saturated sign. 

108 

109 Includes anti-windup back‑calculation: the integrator state ``z`` is 

110 updated using the difference between the saturated and unsaturated 

111 control multiplied by ``Kaw``【789743582768797†L224-L249】. Returns 

112 ``(u_saturated, new_z, sigma)``. 

113 """ 

114 # Continuous term and disturbance‑like internal state 

115 u_cont = -alg_gain_K1 * np.sqrt(np.abs(sigma)) * sgn_sigma 

116 u_dis = z 

117 # Compute unsaturated control 

118 u_raw = u_eq + u_cont + u_dis - damping_gain * sigma 

119 # Saturate the control 

120 u_sat = u_raw 

121 if u_sat > max_force: 

122 u_sat = max_force 

123 elif u_sat < -max_force: 

124 u_sat = -max_force 

125 # Anti-windup back‑calculation: adjust integrator when saturation occurs 

126 new_z = z - alg_gain_K2 * sgn_sigma * dt + Kaw * (u_sat - u_raw) * dt 

127 # Saturate the integrator to maintain boundedness 

128 if new_z > max_force: 

129 new_z = max_force 

130 elif new_z < -max_force: 

131 new_z = -max_force 

132 return float(u_sat), float(new_z), float(sigma)

135class SuperTwistingSMC: 

136 """ 

137 Second‑order (super‑twisting) sliding‑mode controller for the double‑inverted pendulum. 

138 

139 This controller implements the super‑twisting algorithm, a higher‑order 

140 sliding‑mode technique that achieves finite‑time convergence without 

141 requiring direct measurement of the sliding surface derivative. Compared 

142 with conventional (first‑order) sliding‑mode control, the super‑twisting 

143 algorithm is known to mitigate chattering and reduce control effort, 

144 offering improved tracking accuracy. The 

145 implementation uses a continuous saturation function within a boundary 

146 layer ``ε`` to approximate the sign of the sliding variable σ, 

147 consistent with the boundary‑layer approach for chattering reduction 

148 . The size of ``ε`` therefore controls 

149 the trade‑off between chattering attenuation and steady‑state accuracy 

150 . 

151 

152 **Gains:** 

153 

154 - If a 2‑element sequence ``[K1, K2]`` is provided, the sliding‑surface 

155 gains and poles ``(k1, k2, λ1, λ2)`` are set to default values. 

156 - A 6‑element sequence ``[K1, K2, k1, k2, λ1, λ2]`` specifies all 

157 super‑twisting and surface parameters explicitly. 

158 

159 **Sliding surface:** 

160 

161 .. math:: 

162 \\sigma = k_1\\,(\\dot{\\theta}_1 + \\lambda_1\\,\\theta_1) + k_2\\,(\\dot{\\theta}_2 + \\lambda_2\\,\\theta_2). 

163 

164 **Discrete‑time control law:** 

165 

166 .. math:: 

167 \\begin{aligned} 

168 u &= u_{\\\\text{eq}} - K_1 \\\\sqrt{|\\\\sigma|}\\\\,\\\\operatorname{sat}\\\\left( \\\\frac{\\\\sigma}{\\\\epsilon} \\\\right) + z - d\\\\,\\\\sigma,\\\\\\\\ 

169 z^+ &= z - K_2\\\\,\\\\operatorname{sat}\\\\left( \\\\frac{\\\\sigma}{\\\\epsilon} \\\\right)\\\\,dt, 

170 \\end{aligned} 

171 

172 where ``sat`` is a continuous approximation of ``sign`` (either linear or 

173 hyperbolic tangent), ``d`` is the optional damping gain and ``u_eq`` is 

174 the equivalent control derived from the dynamics model. The final output 

175 ``u`` and the disturbance‑like internal state ``z`` are both saturated 

176 to lie within the actuator limits. 

177 

178 The boundary layer ε (> 0) is validated at construction time to avoid 

179 division by zero in the saturated sign computation. A well‑chosen ε 

180 ensures finite‑time convergence and reduces chattering. 

181 

182 **Gain positivity (F‑4.SMCDesign.2 / RC‑04)**: For finite‑time convergence of 

183 the super‑twisting algorithm the algorithmic gains ``K1`` and ``K2`` must 

184 be strictly positive and the sliding‑surface gains ``k1`` and ``k2`` together 

185 with the slope parameters ``λ1`` and ``λ2`` must also be strictly positive. 

186 Super‑twisting literature emphasises that positive constants are required to 

187 ensure robust finite‑time stability【MorenoOsorio2012†L27-L40】 and positive 

188 sliding‑surface coefficients guarantee that the error terms combine with 

189 positive weights【OkstateThesis2013†L1415-L1419】. The constructor therefore 

190 validates all gains using ``require_positive`` and raises a ``ValueError`` 

191 when any gain is non‑positive. 

192 

193 Returns 

194 ------- 

195 tuple 

196 A triple ``(u, (z, σ), history)`` containing the saturated control 

197 signal ``u``, the updated internal state and sliding surface value, 

198 and a history dictionary (empty for this controller). 

199 """

399 def validate_gains(self, gains_b: "np.ndarray") -> "np.ndarray": 

400 """ 

401 Vectorized feasibility check for super‑twisting SMC gains. 

402 

403 The algorithmic gains ``K1`` and ``K2`` must be strictly positive to 

404 ensure finite‑time convergence of the super‑twisting algorithm. Additionally, 

405 for stability, K1 > K2 is required per super-twisting theory. When 

406 a six‑element gain vector is provided the sliding‑surface gains 

407 ``k1``, ``k2`` and the lambda parameters ``lam1``, ``lam2`` must also 

408 be positive. Positive sliding‑surface coefficients are required 

409 because the sliding surface is a linear combination of state 

410 variables with positive weights. 

411 

412 Parameters 

413 ---------- 

414 gains_b : np.ndarray 

415 Array of shape (B, D) containing candidate gain vectors. The 

416 first two columns correspond to ``K1`` and ``K2``; if ``D`` ≥ 6 

417 then columns 3–6 correspond to ``k1``, ``k2``, ``lam1`` and 

418 ``lam2`` respectively. 

419 

420 Returns 

421 ------- 

422 np.ndarray 

423 Boolean mask of shape (B,) indicating which rows satisfy the 

424 positivity and stability constraints. 

425 """ 

426 import numpy as _np 

427 if gains_b.ndim != 2 or gains_b.shape[1] < 2: 

428 return _np.ones(gains_b.shape[0], dtype=bool) 

429 # Always require K1 and K2 to be positive AND K1 > K2 for stability 

430 k1 = gains_b[:, 0].astype(float) 

431 k2 = gains_b[:, 1].astype(float) 

432 valid = (k1 > 0.0) & (k2 > 0.0) & (k1 > k2) 

433 # If sliding surface parameters are provided, require them to be positive 

434 if gains_b.shape[1] >= 6: 

435 surf_k1 = gains_b[:, 2].astype(float) 

436 surf_k2 = gains_b[:, 3].astype(float) 

437 lam1 = gains_b[:, 4].astype(float) 

438 lam2 = gains_b[:, 5].astype(float) 

439 valid = valid & (surf_k1 > 0.0) & (surf_k2 > 0.0) & (lam1 > 0.0) & (lam2 > 0.0) 

440 return valid

531 def _compute_equivalent_control(self, state: np.ndarray) -> float: 

532 """Compute the model‑based equivalent control ``u_eq`` using Tikhonov regularisation. 

533 

534 The original implementation used an SVD‑based condition estimate and a 

535 pseudo‑inverse to invert the inertia matrix. This version instead 

536 applies a small diagonal regularisation and solves the resulting 

537 linear systems directly, reducing computational overhead and 

538 eliminating the need for a full SVD. Adding a small constant to 

539 the diagonal of a symmetric indefinite matrix makes it positive 

540 definite and invertible, a standard technique in control 

541 literature. 

542 """ 

543 if self.dyn is None: 

544 return 0.0 

545 

546 q_dot = state[3:] 

547 try: 

548 M, C, G = self.dyn._compute_physics_matrices(state) 

549 except Exception as e: 

550 logging.warning(f"Physics matrix computation failed, returning safe zero control: {e}") 

551 return 0.0 # OK: Safe fallback when dynamics unavailable 

552 

553 # Regularise the inertia matrix using the configurable constant. A 

554 # small diagonal offset guarantees invertibility of symmetric 

555 # matrices. 

556 M_reg = M + np.eye(3) * max(self.regularization, 0.0) 

557 try: 

558 # Compute controllability scalar L M^-1 B. Use the boundary layer 

559 # thickness as a threshold for ill‑conditioned feedforward terms. 

560 Minv_B = np.linalg.solve(M_reg, self.B) 

561 L_Minv_B = float(self.L @ Minv_B) 

562 # If the equivalent control would divide by a very small number, 

563 # return zero. The threshold equals the boundary layer epsilon 

564 # such that the same tuning parameter governs both chattering 

565 # mitigation and model invertibility. 

566 if abs(L_Minv_B) < self.boundary_layer: 

567 return 0.0 

568 

569 # Compute rhs = C q_dot + G. Support both matrix and vector forms. 

570 if getattr(C, "ndim", 1) == 2: 

571 rhs = C @ q_dot + G 

572 else: 

573 rhs = C + G 

574 Minv_rhs = np.linalg.solve(M_reg, rhs) 

575 num = float(self.L @ Minv_rhs) 

576 # Equivalent control is the feedforward term minus the damping 

577 # portion from the sliding surface definition. Following 

578 # conventional SMC, subtract the inner product of the surface 

579 # gains with the joint velocities. Note the sign convention 

580 # matches the one used in ClassicalSMC. 

581 u_eq = (num - (self.surf_gain_k1 * self.surf_lam1 * q_dot[1] + 

582 self.surf_gain_k2 * self.surf_lam2 * q_dot[2])) / L_Minv_B 

583 except np.linalg.LinAlgError: 

584 # Singular matrix despite regularisation; disable equivalent control 

585 return 0.0 

586 

587 # Do not apply additional clamping here. The final saturation in 

588 # ``compute_control`` will bound the control effort. Returning the 

589 # raw equivalent control maintains transparency and avoids 

590 # inadvertently biasing the controller output. 

591 return float(u_eq)
