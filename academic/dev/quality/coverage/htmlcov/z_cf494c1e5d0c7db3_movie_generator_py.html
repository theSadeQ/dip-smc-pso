150 def _compile_scenes_to_movie(self, scenes: List[MovieScene], output_path: str, fps: int) -> str: 

151 """Compile all scenes into a single movie file.""" 

152 

153 # Create individual scene animations 

154 scene_files = [] 

155 

156 for i, scene in enumerate(scenes): 

157 scene_path = os.path.join(self.output_dir, f"scene_{i:02d}_{scene.title.replace(' ', '_')}.mp4") 

158 

159 if scene.scene_type == 'animation': 

160 self._create_animation_scene(scene, scene_path, fps) 

161 elif scene.scene_type == 'static': 

162 self._create_static_scene(scene, scene_path, fps) 

163 elif scene.scene_type == 'comparison': 

164 self._create_comparison_scene(scene, scene_path, fps) 

165 elif scene.scene_type == 'analysis': 

166 self._create_analysis_scene(scene, scene_path, fps) 

167 

168 scene_files.append(scene_path) 

169 

170 # Compile scenes using ffmpeg (if available) 

171 try: 

172 self._merge_videos(scene_files, output_path) 

173 except Exception as e: 

174 print(f"Video compilation failed: {e}") 

175 print("Individual scene files created successfully:") 

176 for file in scene_files: 

177 print(f" - {file}") 

178 

179 return output_path

181 def _create_animation_scene(self, scene: MovieScene, output_path: str, fps: int) -> None: 

182 """Create an animated scene.""" 

183 data = scene.data 

184 

185 # Extract simulation data 

186 time_history = data.get('time', []) 

187 state_history = data.get('states', []) 

188 control_history = data.get('controls', []) 

189 

190 if not all([time_history, state_history, control_history]): 

191 print(f"Warning: Insufficient data for animation scene '{scene.title}'") 

192 return 

193 

194 # Create animation 

195 animator = DIPAnimator(None, figsize=(16, 9)) 

196 

197 # Add title and description 

198 animator.fig.suptitle(scene.title, fontsize=20, fontweight='bold') 

199 animator.ax.text(0.02, 0.02, scene.description, transform=animator.ax.transAxes, 

200 fontsize=12, bbox=dict(boxstyle="round", facecolor="white", alpha=0.9)) 

201 

202 # Save animation 

203 animator.save_animation(state_history, control_history, time_history, output_path, fps=fps)

263 """Create an analysis visualization scene."""

267 fig = plt.figure(figsize=(16, 9))

281 def animate_analysis(frame): 

282 # Gradually reveal analysis results 

283 alpha = min(1.0, frame / (frames * 0.3)) 

284 for ax in fig.get_axes(): 

285 for artist in ax.get_children(): 

286 if hasattr(artist, 'set_alpha'): 

287 artist.set_alpha(alpha) 

288 return fig.get_children() 

289 

290 anim = animation.FuncAnimation(fig, animate_analysis, frames=frames, interval=1000/fps) 

291 anim.save(output_path, writer='pillow', fps=fps)

325 def _create_statistical_analysis(self, fig: plt.Figure, data: Dict[str, Any]) -> None: 

326 """Create statistical analysis plots.""" 

327 gs = fig.add_gridspec(2, 2) 

328 

329 # Performance distributions 

330 ax1 = fig.add_subplot(gs[0, :]) 

331 if 'performance_distributions' in data: 

332 for controller, values in data['performance_distributions'].items(): 

333 ax1.hist(values, alpha=0.7, label=controller, bins=30) 

334 ax1.set_title("Performance Distributions") 

335 ax1.set_xlabel("Performance Metric") 

336 ax1.set_ylabel("Frequency") 

337 ax1.legend() 

338 ax1.grid(True, alpha=0.3) 

339 

340 # Confidence intervals 

341 ax2 = fig.add_subplot(gs[1, 0]) 

342 if 'confidence_intervals' in data: 

343 controllers = list(data['confidence_intervals'].keys()) 

344 means = [data['confidence_intervals'][c]['mean'] for c in controllers] 

345 errors = [data['confidence_intervals'][c]['error'] for c in controllers] 

346 ax2.errorbar(range(len(controllers)), means, yerr=errors, fmt='o', capsize=5) 

347 ax2.set_xticks(range(len(controllers))) 

348 ax2.set_xticklabels(controllers, rotation=45) 

349 ax2.set_title("Performance Confidence Intervals") 

350 ax2.grid(True, alpha=0.3)

359 def _merge_videos(self, video_files: List[str], output_path: str) -> None: 

360 """Merge multiple video files using ffmpeg.""" 

361 import subprocess 

362 

363 # Create file list for ffmpeg 

364 list_file = os.path.join(self.output_dir, "video_list.txt") 

365 with open(list_file, 'w') as f: 

366 for video_file in video_files: 

367 f.write(f"file '{video_file}'\n") 

368 

369 # Run ffmpeg 

370 cmd = [ 

371 'ffmpeg', '-f', 'concat', '-safe', '0', '-i', list_file, 

372 '-c', 'copy', output_path, '-y' 

373 ] 

374 

375 subprocess.run(cmd, check=True) 

376 os.remove(list_file)
