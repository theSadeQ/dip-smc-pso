5""" 

6Expose the top‑level ``src`` package and provide convenient access 

7to its submodules. 

8 

9This module exists to ensure that the ``src`` directory is treated as 

10a proper Python package during test collection. Without an 

11``__init__.py`` file, pytest may fail to locate attributes such as 

12``src.config`` when performing monkeypatching. 

13 

14In addition to making the package importable, we deliberately expose 

15the ``config`` submodule as an attribute on the package. Some tests 

16refer to ``src.config`` directly when patching configuration loaders 

17via ``monkeypatch.setattr``. Importing the submodule here ensures 

18that ``src.config`` resolves to the actual configuration module, 

19rather than raising ``AttributeError: module 'src' has no attribute 

20'config'`` during test execution. 

21 

22We avoid importing any other submodules by default to prevent 

23side effects or unnecessary dependencies from being loaded at 

24package import time. Users should import submodules explicitly as 

25needed. 

26""" 

27 

28from importlib import import_module 

29# ----------------------------------------------------------------------------- 

30# Warning filters 

31# 

32# Some third‑party dependencies emit noisy warnings during normal operation. 

33# Suppress specific warnings here to keep test output clean while preserving 

34# other warnings. See pytest warnings summary for details. 

35import warnings 

36 

37# Suppress matplotlib animation warning when an animation object is garbage 

38# collected before being rendered. The Visualizer stores the animation on 

39# itself, but if the calling code does not retain a reference, matplotlib 

40# emits a UserWarning. This filter silences that warning globally. 

41warnings.filterwarnings( 

42 "ignore", 

43 message="Animation was deleted without rendering anything", 

44 category=UserWarning, 

45 module="matplotlib.animation", 

46) 

47 

48# Suppress a PendingDeprecationWarning from the osqp solver interface. The 

49# default value of ``raise_error`` will change in a future release; our 

50# MPCController explicitly handles solver failure by falling back, so we 

51# choose to silence this noisy warning. 

52warnings.filterwarnings( 

53 "ignore", 

54 category=PendingDeprecationWarning, 

55 module="osqp.interface", 

56) 

57 

58# Expose the configuration module as an attribute. Importing 

59# ``src.config`` here ensures that monkeypatching via ``src.config`` 

60# operates on the correct module. See tests that call 

61# ``monkeypatch.setattr('src.config.load_config', ...)``. 

62try: 

63 config = import_module('.config', __package__) 

64except Exception: 

65 # If the configuration module fails to import (e.g., due to 

66 # intentionally missing dependencies in a test environment), define a 

67 # placeholder object so that ``src.config`` always exists. The 

68 # placeholder allows attribute assignment via monkeypatching but will 

69 # otherwise raise the original import exception when attributes are 

70 # accessed. This avoids AttributeError: module 'src' has no 

71 # attribute 'config' while preserving the original failure mode. 

72 import types 

73 

74 class _ConfigPlaceholder(types.SimpleNamespace): 

75 # Capture the exception to raise when an undefined attribute is 

76 # accessed. ``exc`` is closed over from the except block. 

77 def __init__(self, exc: Exception): 

78 super().__init__() 

79 self.__exc = exc 

80 

81 def __getattr__(self, item): 

82 # When any attribute other than those explicitly set via 

83 # monkeypatching is accessed, re-raise the original import 

84 # exception to signal that the real config could not be 

85 # imported. This mirrors the behavior one would see if 

86 # ``src.config`` failed to import normally. 

87 raise self.__exc 

88 

89 # Use the caught exception from the outer scope to initialize the 

90 # placeholder. ``exc_info()`` is not available here so reuse 

91 # Exception from except. 

92 import sys 

93 # Capture the current exception info. sys.exc_info()[1] holds the 

94 # exception instance that was raised when attempting to import 

95 # src.config. If no exception is available (should not happen 

96 # because we are in an except block), default to a generic 

97 # ImportError. 

98 exc = sys.exc_info()[1] or ImportError("src.config could not be imported") 

99 config = _ConfigPlaceholder(exc) 

100 

101# Export only the explicitly exposed names from this package. Keeping 

102# ``__all__`` small avoids accidentally re‑exporting every submodule. 

103__all__ = ['config']
