138 timestamp = datetime.now()

140 with self._lock:

146 def update_metrics(self, metrics: Dict[str, float], timestamp: Optional[datetime] = None): 

147 """Update multiple metrics at once.""" 

148 if timestamp is None: 

149 timestamp = datetime.now() 

150 

151 with self._lock: 

152 for name, value in metrics.items(): 

153 if name not in self.metrics: 

154 self.register_metric(name) 

155 self.metrics[name].values.append((timestamp, value))

157 def get_metric_data(self, name: str, duration_minutes: int = 60) -> List[Tuple[datetime, float]]: 

158 """Get metric data for the last N minutes.""" 

159 if name not in self.metrics: 

160 return [] 

161 

162 cutoff = datetime.now() - timedelta(minutes=duration_minutes) 

163 return [ 

164 (timestamp, value) 

165 for timestamp, value in self.metrics[name].values 

166 if timestamp >= cutoff

535 async def start_server(self): 

536 """Start the dashboard web server.""" 

537 try: 

538 from aiohttp import web, web_runner 

539 import aiohttp_cors 

540 

541 app = web.Application() 

542 

543 # Add CORS support 

544 cors = aiohttp_cors.setup(app, defaults={ 

545 "*": aiohttp_cors.ResourceOptions( 

546 allow_credentials=True, 

547 expose_headers="*", 

548 allow_headers="*", 

549 allow_methods="*" 

550 ) 

551 }) 

552 

553 # Routes 

554 app.router.add_get('/', self._handle_dashboard) 

555 app.router.add_get('/api/metrics', self._handle_metrics_api) 

556 app.router.add_get('/api/export', self._handle_export_api) 

557 app.router.add_get('/api/layouts', self._handle_layouts_api) 

558 app.router.add_post('/api/layouts', self._handle_create_layout_api) 

559 

560 # Add CORS to all routes 

561 for route in list(app.router.routes()): 

562 cors.add(route) 

563 

564 runner = web_runner.AppRunner(app) 

565 await runner.setup() 

566 

567 site = web_runner.TCPSite(runner, self.host, self.port) 

568 await site.start() 

569 

570 self.is_running = True 

571 self.logger.info(f"Dashboard server started at http://{self.host}:{self.port}") 

572 

573 # Open browser 

574 url = f"http://{self.host}:{self.port}" 

575 self.logger.info(f"Opening dashboard in browser: {url}") 

576 webbrowser.open(url) 

577 

578 except Exception as e: 

579 self.logger.error(f"Failed to start dashboard server: {e}") 

580 raise

588 async def _handle_metrics_api(self, request): 

589 """Handle metrics API request.""" 

590 from aiohttp import web 

591 

592 duration_minutes = int(request.query.get('duration', 60)) 

593 metrics_data = {} 

594 

595 for name, series in self.metrics.items(): 

596 data = self.get_metric_data(name, duration_minutes) 

597 if data: 

598 timestamps = [ts.isoformat() for ts, _ in data] 

599 values = [val for _, val in data] 

600 metrics_data[name] = { 

601 'timestamps': timestamps, 

602 'values': values, 

603 'unit': series.unit, 

604 'description': series.description 

605 } 

606 

607 return web.json_response(metrics_data)
